import torch
import torch.nn as nn
import torch.nn.functional as F
from torchvision.models import resnet50
import clip
from transformers import BertModel

class ECommerceMultimodalModel(nn.Module):
    def __init__(
        self,
        bert_model=None,
        bert_name="bert-base-chinese",
        clip_name="ViT-B/32",
        hidden_dim=256,
        num_classes=2,
        device="cpu"
    ):
        super().__init__()
        self.device = device

        # ===== 0. ResNet 视觉增强分支 (修改：解冻最后两层) =====
        self.resnet = resnet50(pretrained=True)
        self.resnet.fc = nn.Identity()
        
        # 首先冻结全部 ResNet 参数
        for p in self.resnet.parameters():
            p.requires_grad = False
        
        resnet_dim = 2048

        # ===== 1. BERT 文本编码器 (保持冻结) =====
        if bert_model is not None:
            self.bert = bert_model
        else:
            self.bert = BertModel.from_pretrained(bert_name)

        for p in self.bert.parameters():
            p.requires_grad = False
        bert_dim = self.bert.config.hidden_size 

        # ===== 2. CLIP (保持冻结) =====
        self.clip_model, _ = clip.load(clip_name, device="cpu")
        for p in self.clip_model.parameters():
            p.requires_grad = False
        clip_dim = self.clip_model.text_projection.shape[1] 

        # ===== 3. 单模态投影 =====
        self.proj_text = nn.Linear(768 + 512, hidden_dim)
        self.proj_img  = nn.Linear(2048 + 512, hidden_dim)
        self.proj_clip = nn.Linear(512 + 512, hidden_dim)

        # ===== 增加：LayerNorm 层 (用于稳定门控前的特征分布) =====
        self.ln_text = nn.LayerNorm(hidden_dim)
        self.ln_img  = nn.LayerNorm(hidden_dim)
        self.ln_clip = nn.LayerNorm(hidden_dim)

        # ===== 4. 门控网络 =====
        self.gate_fc = nn.Linear(hidden_dim * 3, 3) 

        # ===== 5. 分类器 =====
        self.classifier = nn.Sequential(
            nn.Linear(hidden_dim, hidden_dim),
            nn.LayerNorm(hidden_dim),
            nn.GELU(),
            nn.Dropout(0.3),
            nn.Linear(hidden_dim, hidden_dim // 2),
            nn.GELU(),
            nn.Dropout(0.2),
            nn.Linear(hidden_dim // 2, num_classes)
        )

    def forward(self,
                bert_input_ids, bert_attention_mask,
                clip_input_ids, image_tensor):
        
        # 1) TextPath: BERT + CLIP-text
        bert_out = self.bert(input_ids=bert_input_ids,
                            attention_mask=bert_attention_mask)
        bert_cls = bert_out.pooler_output
        clip_text = self.clip_model.encode_text(clip_input_ids)
        clip_text = F.normalize(clip_text, dim=-1)

        Ft = self.proj_text(torch.cat([bert_cls, clip_text], dim=1))
        # 修改：引入 LayerNorm
        Ft = self.ln_text(Ft)

        # 2) ImagePath: ResNet + CLIP-image
        resnet_vis = self.resnet(image_tensor)
        clip_vis = self.clip_model.encode_image(image_tensor)
        clip_vis = F.normalize(clip_vis, dim=-1)

        Fi = self.proj_img(torch.cat([resnet_vis, clip_vis], dim=1))
        # 修改：引入 LayerNorm
        Fi = self.ln_img(Fi)

        # 3) CLIP 跨模态融合
        Fm = self.proj_clip(torch.cat([clip_text, clip_vis], dim=1))
        # 修改：引入 LayerNorm
        Fm = self.ln_clip(Fm)

        # 4) 余弦相似度门控
        sim = F.cosine_similarity(clip_text, clip_vis, dim=1).unsqueeze(1)
        Fm_weighted = sim * Fm

        # 5) 动态三系数
        gate_input = torch.cat([Ft, Fi, Fm_weighted], dim=1)
        alpha = F.softmax(self.gate_fc(gate_input), dim=1)

        F_final = (alpha[:, 0:1] * Ft +
                alpha[:, 1:2] * Fi +
                alpha[:, 2:3] * Fm_weighted)

        # 6) 分类
        logits = self.classifier(F_final)
        return logits
